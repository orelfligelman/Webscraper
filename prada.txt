    			 filteredByType = filterAssociative.type.toLowerCase() + "&nbsp;/&nbsp;";
    		 } else {
    			 // questo caso e' piu' complesso, ho i filtri multipli
    			 filteredByType = filterAssociative.type.split(":")[0].toLowerCase() + " .../&nbsp;";
    		 }
    	 }
    	 if(filterAssociative.material != "" && filterAssociative.material != undefined) {
             if(filterAssociative.material.indexOf(":") === -1) {
                 filteredByMaterial = filterAssociative.material.toLowerCase() + "&nbsp;/&nbsp;";
             } else {
                 // questo caso e' piu' complesso, ho i filtri multipli
                 filteredByMaterial = filterAssociative.material.split(":")[0].toLowerCase() + " .../&nbsp;";
             }
         }
    	 if(filterAssociative.color != "" && filterAssociative.color != undefined) {
             if(filterAssociative.color.indexOf(":") === -1) {
                 filteredByColor = filterAssociative.color.toLowerCase() + "&nbsp;/&nbsp;";
             } else {
                 // questo caso e' piu' complesso, ho i filtri multipli
                 filteredByColor = filterAssociative.color.split(":")[0].toLowerCase() + " .../&nbsp;";
             }
         }
    	 if(filterAssociative.size != "" && filterAssociative.size != undefined) {
             if(filterAssociative.size.indexOf(":") === -1) {
                 filteredBySize = filterAssociative.size.toLowerCase() + "&nbsp;/&nbsp;";
             } else {
                 // questo caso e' piu' complesso, ho i filtri multipli
                 filteredBySize = filterAssociative.size.split(":")[0].toLowerCase() + " .../&nbsp;";
             }
         }
    	 
        var filteredBy = filteredByType + filteredByMaterial + filteredByColor + filteredBySize;
        
        // controllo se nell'ultimo carattere e' presente il ; (infatti potrei avere .../&nbsp;)
        if(filteredBy.substring(filteredBy.length - 1) === ";") {
            // elimino l'eventual ultimo /
            filteredBy = filteredBy.substring(0, filteredBy.lastIndexOf("/"));
        }
        
        return filteredBy;
    }
	   
	function createAndUpdateArrayFilter(arrayFilter, nextListElement, nextCombValue01, nextCombName01, nextCombValue02, nextCombName02, nextCombValue03, nextCombName03) {
	        if(nextListElement === undefined || nextListElement === "") {
	            // non ha senso proseguire
	            return;
	        }
	        
	        // catturo l'array esistente con tutti i valori possibili
	        var arrayTmp = arrayFilter.list;
	        
	        // e se e' necessario lo aggiorniamo con il nuovo elemento
	        if(jQuery.inArray(nextListElement, arrayTmp) === -1) {
	            arrayTmp.push(nextListElement);
	            
	            arrayFilter.list = arrayTmp;
	        }
	        
	        var combination = arrayFilter.combination;
	            
	        if(combination[nextListElement] === undefined) {
	            // e' la prima volta che creo le associazioni
	            var nextCombArray01 = new Array();
	            var nextCombArray02 = new Array();
	            var nextCombArray03 = new Array();
	            
	            if(nextCombValue01 != undefined && nextCombValue01 != "") {
	                nextCombArray01.push(nextCombValue01);
	            }
	            if(nextCombValue02 != undefined && nextCombValue02 != "") {
	                nextCombArray02.push(nextCombValue02);
	            }
	            if(nextCombValue03 != undefined && nextCombValue03 != "") {
	                nextCombArray03.push(nextCombValue03);
	            }
	            
	            // create
	            // arrayFilter.combination[nextListElement] = {nextCombName01 : nextCombArray01, nextCombName02 : nextCombArray02, nextCombName03 : nextCombArray03};
	            arrayFilter.combination[nextListElement] = new Array();
	            arrayFilter.combination[nextListElement][nextCombName01] = nextCombArray01;
	            arrayFilter.combination[nextListElement][nextCombName02] = nextCombArray02;
	            arrayFilter.combination[nextListElement][nextCombName03] = nextCombArray03;
	            
	        } else {
	            // esiste gia' l'elemento nella lista, devo aggiornare le combinazioni
				var nextCombArray01 = arrayFilter.combination[nextListElement][nextCombName01];
				var nextCombArray02 = arrayFilter.combination[nextListElement][nextCombName02];
				var nextCombArray03 = arrayFilter.combination[nextListElement][nextCombName03];
				
				if(jQuery.inArray(nextCombValue01, nextCombArray01) === -1 && nextCombValue01 != undefined) {
				    nextCombArray01.push(nextCombValue01);
				}
				if(jQuery.inArray(nextCombValue02, nextCombArray02) === -1 && nextCombValue02 != undefined) {
				    nextCombArray02.push(nextCombValue02);
				}
				if(jQuery.inArray(nextCombValue03, nextCombArray03) === -1 && nextCombValue03 != undefined) {
				    nextCombArray03.push(nextCombValue03);
				}
				
				// update
				// arrayFilter.combination[nextListElement] = {nextCombName01 : nextCombArray01, nextCombName02 : nextCombArray02, nextCombName03 : nextCombArray03};
				arrayFilter.combination[nextListElement][nextCombName01] = nextCombArray01;
				arrayFilter.combination[nextListElement][nextCombName02] = nextCombArray02;
				arrayFilter.combination[nextListElement][nextCombName03] = nextCombArray03;
	        }
	    }

	    function productMatchWithFilter(typeProduct, materialProduct, colorProduct, sizeProduct) {
	    
	        var numberOfProperties = 0;
	        
	        for(var i = 0; i < filterTypeActive.length; i++) {
	            switch(filterTypeActive[i]) {
	                case "type":
	                	var filterAssociativeTypeSplit = filterAssociative.type.split(":");
	                    if(jQuery.inArray(typeProduct, filterAssociativeTypeSplit) !== -1) {  
	                        numberOfProperties++;
	                    }
	                    break;
	                case "material":
	                	var filterAssociativeMaterialSplit = filterAssociative.material.split(":");
	                    if(jQuery.inArray(materialProduct, filterAssociativeMaterialSplit) !== -1) { 
	                        numberOfProperties++;
	                    }
	                    break;
	                case "color":
	                	var filterAssociativeColorSplit = filterAssociative.color.split(":");
	                    if(jQuery.inArray(colorProduct, filterAssociativeColorSplit) !== -1) { 
	                        numberOfProperties++;
	                    }
	                    break;
	                case "size":
	                	var filterAssociativeSizeSplit = filterAssociative.size.split(":");
	                    if(jQuery.inArray(sizeProduct, filterAssociativeSizeSplit) !== -1) { 
	                        numberOfProperties++;
	                    }
	                    break;
	                default:
	                    break;
	            }
	        }
	        return (numberOfProperties == filterTypeActive.length);
	    }
